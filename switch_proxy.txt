function enableproxy()
{	
################changing in environment########################
	sudo bash -c 'echo PATH=\"/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin:/usr/games:/usr/local/games\" > /etc/environment'
	sudo bash -c 'echo Acquire::http::proxy \"http://134.245.6.93:80\">>/etc/environment'
	sudo bash -c 'echo Acquire::https::proxy \"http://134.245.6.93:80\">>/etc/environment'
#	sudo bash -c 'echo Acquire::ftp::proxy \"ftp://134.245.6.93:80\">>/etc/environment'
#	sudo bash -c 'echo Acquire::socks::proxy \"socks://134.245.6.93:80\">>/etc/environment'
###############changing in apt conf.d ######################
	sudo bash -c 'echo Acquire::http::proxy \"http://134.245.6.93:80\"\;>/etc/apt/apt.conf.d/01proxy'
	sudo bash -c 'echo Acquire::https::proxy \"https://134.245.6.93:80\"\;>>/etc/apt/apt.conf.d/01proxy'
#	sudo bash -c 'echo Acquire::ftp::proxy \"ftp://172.25.6.6:3128\"\;>>/etc/apt/apt.conf.d/01proxy'
